11     * 1. "As a user of your command-line program, I should be able to search a CSV file for rows that
12     * contain a value I provide in a given column, or all columns.‚Äù
13     *
14     * <p>utility program that takes in csv filename, value to search for, and a column identifier the
15     * column identifier can be an index or a column name if csv file has column headers (boolean)
16     * (may have the user specify whether the csv contains headers) if no column identifier, the
17     * search should be performed on all columns
18     *
19     * <p>Should print the rows which contain the given value (in the column of interest if there is
20     * one)
21     *
22     * <p>Note: make sure to handle upper, lower, or mismatches if the file cannot be found or opened,
23     * or an error is encountered when reading, print an appropriate error message via
24     * System.err.println and terminate rather than crashing with an exception (Design) where should
25     * these messages be created? should CSV parser class itself be printing to the screen? Consider
26     * different possible use-cases
27     */
28    public CSVSearcher(
29        CSVParser parser, String searchTerm, Boolean hasHeaders, String columnIdentifier)
30        throws IOException, FactoryFailureException {
31  
32      List<List<String>> parsed = parser.parse();
33      parser.close();
34  
35      List<List<String>> result;
36      // if there is no column identifier, then search through every column of each row to find
37      // and return the rows that have the search term
38      if (columnIdentifier == null) {
39  
40        result = this.searchAll(parsed, searchTerm, hasHeaders);
41        if (checkEmpty(result)) {
42          System.out.printf("your search term does not exist");
43        } else {
44          this.print(result);
45        }
46      }
47      // if it has a column name
48      else if (!columnIdentifier.matches("\\d+")) {
49        int col = -1;
50        List<String> row = parsed.get(0);
51        for (int i = 0; i < row.size(); i++) {
52          if ((row.get(i)).equals(columnIdentifier)) {
53            col = i;
54            break;
55          }
56        }
57        if (col == -1) {
58          throw new IOException("the inputted column identifier name does not exist");
59        }
60  
61        result = this.searchCol(parsed, searchTerm, hasHeaders, col);
62        if (checkEmpty(result)) {
63          System.out.printf("your search term does not exist in this column");
64        } else {
65          this.print(result);
66        }
67  
68      } else {
69        int col = Integer.parseInt(columnIdentifier);
70  
71        if (col > parsed.size() - 1 || col < 0) {
72          throw new IOException("the inputted column identifier index does not exist");
73        }
74  
75        result = this.searchCol(parsed, searchTerm, hasHeaders, col);
76        if (checkEmpty(result)) {
77          System.out.printf("your search term does not exist in this column");
78        } else {
79          this.print(result);
80        }
81      }
82    }

84    public List<List<String>> searchAll(
85        List<List<String>> parsed, String searchTerm, boolean hasHeaders) {
86      List<List<String>> result = new ArrayList<>();
87  
88      // if it has headers, then remove the headers row
89      if (hasHeaders) {
90        parsed.remove(0);
91      }
92  
93      for (int i = 0; i < parsed.size(); i++) {
94        List<String> row = parsed.get(i);
95  
96        for (int j = 0; j < row.size(); j++) {
97          if (row.get(j).equals(searchTerm)) {
98            result.add(row);
99            break;
100         }
101       }
102     }
103 
104     return result;
105   }

107   public List<List<String>> searchCol(
108       List<List<String>> parsed, String searchTerm, boolean hasHeaders, int col) {
109     List<List<String>> result = new ArrayList<>();
110 
111     // if it has headers, then remove the headers row
112     if (hasHeaders) {
113       parsed.remove(0);
114     }
115 
116     for (int i = 0; i < parsed.size(); i++) {
117       List<String> row = parsed.get(i);
118       if (row.get(col).equals(searchTerm)) {
119         result.add(row);
120       }
121     }
122 
123     return result;
124   }

126   public void print(List<List<String>> result) {
127 
128     for (int i = 0; i < result.size(); i++) {
129       List<String> innerList = result.get(i);
130       for (int j = 0; j < innerList.size(); j++) {
131         String element = innerList.get(j);
132         System.out.print(element + " ");
133       }
134       System.out.println(); // Newline for separating inner lists
135     }
136   }

138   public boolean checkEmpty(List<List<String>> result) {
139     for (List<String> innerList : result) {
140       if (!innerList.isEmpty()) {
141         return false;
142       }
143     }
144     return true;
