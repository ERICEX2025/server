5   import java.io.BufferedReader;
6   import java.io.IOException;
7   import java.io.Reader;
8   import java.util.ArrayList;
9   import java.util.List;
10  import java.util.regex.Pattern;
11  
12  public class CSVParser<T> {
13  
14    private final BufferedReader reader;
15    private final CreatorFromRow<T> rowCreator;
16    static final Pattern regexSplitCSVRow =
17        Pattern.compile(",(?=([^\\\"]*\\\"[^\\\"]*\\\")*(?![^\\\"]*\\\"))");
18  
19    public CSVParser(Reader reader, CreatorFromRow<T> rowCreator) {
20      this.reader = new BufferedReader(reader);
21      this.rowCreator = rowCreator;
22    }
23  
24    public List<T> parse() throws IOException, FactoryFailureException {
25      List<T> result = new ArrayList<>();
26      String line;
27      while ((line = reader.readLine()) != null) {
28        System.out.println(line);
29        System.out.println("ERICCCCCCCCC");
30        System.out.println(result.size());
31        List<String> row = arrayToList(regexSplitCSVRow.split(line));
32        T dataObject = rowCreator.create(row);
33        result.add(dataObject);
34      }
35      if (result.isEmpty()) {
36        throw new IOException("Empty CSV file ");
37      }
38      return result;
39    }
40  
41    public void close() throws IOException {
42      reader.close();
43    }
44  
45    // Helper method to convert String[] to List<String>
46    private static List<String> arrayToList(String[] array) {
47      List<String> list = new ArrayList<>();
48      for (String str : array) {
49        list.add(str);
50      }
51      return list;
52    }
53  }
